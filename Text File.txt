'use client'; import { useCallback, useRef } from 'react';
/**
    • react-force-graph-2d renders to a single , so node/edge "animation"
    • has to be painted manually inside nodeCanvasObject / linkCanvasObject on
    • every tick, rather than via CSS or Framer. This module provides:
        ◦ useNodeRipples(): tracks active click-ripples and exposes addRipple()
        ◦ paintNode(): pulsing glow + label, call from 
        ◦ paintLink(): animated flowing dash from source -> target, call from
    • Usage:
    • const { ripples, addRipple } = useNodeRipples();
    • <ForceGraph2D
    • onNodeClick={(node) => addRipple(node)}
    • nodeCanvasObject={(node, ctx, scale) => paintNode(node, ctx, scale, ripples)}
    • linkCanvasObject={(link, ctx) => paintLink(link, ctx)}
    • linkCanvasObjectMode={() => 'replace'}
    • /> */
const TYPE_COLORS = { person: '#FF335C', event: '#FF8A65', location: '#7CFFB2', organization: '#FFD166', fact: '#E8E8F0', document: '#8A8A99', phone: '#FF0033', note: '#B8B8C8', default: '#FF0033', };
export function useNodeRipples() { const ripples = useRef([]); // [{ x, y, start }]
const addRipple = useCallback((node) => { if (node?.x == null) return; ripples.current.push({ x: node.x, y: node.y, start: performance.now() }); // keep the list small if (ripples.current.length > 12) ripples.current.shift(); }, []);
return { ripples, addRipple }; }
const RIPPLE_DURATION = 900; // ms, matches tailwind ripple keyframe timing
/** Draw a single pulsing node with its label. Call once per node per frame. */ export function paintNode(node, ctx, globalScale, ripplesRef) { const r = 5 + (node.weight || 0) * 0.6; const color = TYPE_COLORS[node.type] || TYPE_COLORS.default; const t = performance.now() / 1000;
// ambient pulse: glow radius breathes slowly, offset by node id so the // whole graph doesn't pulse in lockstep const phase = (node.id ? hashSeed(node.id) : 0) % 1000; const pulse = 0.5 + 0.5 * Math.sin(t * 1.6 + phase);
ctx.save(); ctx.shadowColor = color; ctx.shadowBlur = 6 + pulse * 10; ctx.fillStyle = color; ctx.beginPath(); ctx.arc(node.x, node.y, r, 0, 2 * Math.PI); ctx.fill(); ctx.restore();
// label, only above a zoom threshold to avoid clutter if (globalScale > 1.1) { ctx.font = ${10 / globalScale}px "JetBrains Mono", monospace; ctx.fillStyle = 'rgba(232,232,240,0.85)'; ctx.textAlign = 'center'; ctx.fillText(node.label || node.name || '', node.x, node.y + r + 9 / globalScale); }
// active ripples belonging to this node's location if (ripplesRef?.current?.length) { const now = performance.now(); ripplesRef.current = ripplesRef.current.filter((rp) => now - rp.start < RIPPLE_DURATION); for (const rp of ripplesRef.current) { const elapsed = (now - rp.start) / RIPPLE_DURATION; const radius = r + elapsed * r * 9; const alpha = 1 - elapsed; ctx.save(); ctx.strokeStyle = rgba(255,0,51,${alpha * 0.7}); ctx.lineWidth = 1.5 / globalScale; ctx.beginPath(); ctx.arc(rp.x, rp.y, radius, 0, 2 * Math.PI); ctx.stroke(); ctx.restore(); } } }
/** Draw a link with small light particles flowing parent -> child. */ export function paintLink(link, ctx) { const src = link.source; const tgt = link.target; if (!src || !tgt || src.x == null || tgt.x == null) return;
ctx.save(); ctx.strokeStyle = 'rgba(255,0,51,0.18)'; ctx.lineWidth = 1; ctx.beginPath(); ctx.moveTo(src.x, src.y); ctx.lineTo(tgt.x, tgt.y); ctx.stroke();
// 2-3 particles per link, looping along the path based on wall-clock time const dx = tgt.x - src.x; const dy = tgt.y - src.y; const dist = Math.hypot(dx, dy); if (dist < 1) { ctx.restore(); return; }
const speed = 0.00035; // fraction of path per ms const now = performance.now(); const particleCount = 2;
for (let i = 0; i < particleCount; i++) { const offset = i / particleCount; let p = ((now * speed + offset) % 1 + 1) % 1; const px = src.x + dx * p; const py = src.y + dy * p;
ctx.beginPath();
ctx.fillStyle = '#FF335C';
ctx.shadowColor = '#FF0033';
ctx.shadowBlur = 4;
ctx.arc(px, py, 1.6, 0, 2 * Math.PI);
ctx.fill();
}
ctx.restore(); }
function hashSeed(id) { const s = String(id); let h = 0; for (let i = 0; i < s.length; i++) h = (h * 31 + s.charCodeAt(i)) % 1000; return h; }
